Fidelity NetBenefits — Trigger PDF Downloads

Paste the following script into the browser console on the Fidelity NetBenefits statements page to automatically click every download icon (with a 1-second delay between clicks). After all downloads have finished, capture the network traffic as a HAR file and upload it above. 

(async () => {
  const delay = (ms) => new Promise(resolve => setTimeout(resolve, ms))
  const icons = Array.from(document.querySelectorAll('pvd-icon[pvd-name="action__download"]'))
  console.log(`Found ${icons.length} download icons. Starting clicks...`)
  for (let i = 0; i < icons.length; i++) {
    icons[i].click()
    console.log(`Clicked icon ${i + 1} of ${icons.length}. Waiting 1 second...`)
    await delay(1000)
  }
  console.log('Finished clicking all icons!')
})()

Charles Schwab — Trigger PDF Downloads

Paste the following script into the browser console on the Charles Schwab statements page to automatically click every download icon (with a 1-second delay between clicks). After all downloads have finished, capture the network traffic as a HAR file and upload it above. 

(async () => {
  const delay = (ms) => new Promise(resolve => setTimeout(resolve, ms))
  const setWithChange = (sel, val) => {
    const se = document.querySelector(sel)
    se.value = val
    const changeEvent = new Event('change', { bubbles: true })
    se.dispatchEvent(changeEvent)
  }
  setWithChange("#date-range-select-id","Last10Years")
  await delay(100)
  for (const lbl of ["Tax Forms","Letters","Reports & Plans","Trade Confirms","Search"]) {
    Array.from(document.querySelectorAll("button")).filter(e=>e.innerText.includes(lbl)).forEach(e=>e.click())
    await delay(100)
  }
  await delay(5000)
  let pageNumber = 1
  while (true) {
    const buttons = Array.from(document.querySelectorAll("button[aria-label='Click to Download PDF']"))
    console.log(`Page #${pageNumber}: Found ${buttons.length} download buttons. Starting clicks...`)
    for (let i = 0; i < buttons.length; i++) {
      buttons[i].click()
      console.log(`Page #${pageNumber}: Clicked button ${i + 1} of ${buttons.length}. Waiting 1 second...`)
      await delay(1000)
    }
    console.log(`Page #${pageNumber}: Finished clicking all buttons!`)
    const nextLink = document.querySelector("#pagination-next-link")
    if (nextLink.parentElement.classList.contains('sdps-hide')) {
      console.log('No more pages, script exiting')
      break
    }
    pageNumber++
    nextLink.click()
    await delay(1000)
  }
})()
